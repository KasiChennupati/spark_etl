32 def cli_loop(self, run_id): 

33 # line_mode can be "bash", "python" or "OFF" 

34 # When line_mode is OFF, you need send explicitly run @@bash or @@python 

35 # to submit a block of code to server 

36 # When line_mode is bash, each line is a bash script 

37 # when line_mode is python, each line is a python script 

38 line_mode = "off" 

39 

40 # if is_waiting_for_response is True, we need to pull server for cli-response.json 

41 # if is_waiting_for_response is False, we are free to enter new command 

42 is_waiting_for_response = True 

43 # command line buffer 

44 cli_lines = [] 

45 cli_wait_prompt = "-/|\\" 

46 cli_wait_prompt_idx = 0 

47 log_filename = None 

48 # commands 

49 # @@log -- all the output will be written to the log file as well 

50 # by default there is not log 

51 # @@nolog -- turn off log 

52 # @@clear -- clear command line buffer 

53 # @@load -- load a script from local file and append to command buffer 

54 # @@bash -- submit a bash script 

55 # @@python -- submit a python script 

56 # @@show -- show the command buffer 

57 # @@pwd -- show driver's current directory 

58 # @@quit -- quit the cli console 

59 

60 command = None 

61 while True: 

62 if not is_waiting_for_response: 

63 if line_mode == "off": 

64 prompt = "> " 

65 elif line_mode == "bash": 

66 prompt = "bash> " 

67 else: 

68 prompt = "python> " 

69 

70 command = input(prompt) 

71 

72 if command == "@@quit": 

73 self.write_cli_request( 

74 run_id, 

75 { 

76 "type": "@@quit", 

77 } 

78 ) 

79 is_waiting_for_response = True 

80 continue 

81 

82 if command == "@@pwd": 

83 self.write_cli_request( 

84 run_id, 

85 { 

86 "type": "@@pwd", 

87 } 

88 ) 

89 is_waiting_for_response = True 

90 continue 

91 

92 if command == "@@bash": 

93 self.write_cli_request( 

94 run_id, 

95 { 

96 "type": "@@bash", 

97 "lines": cli_lines 

98 } 

99 ) 

100 is_waiting_for_response = True 

101 cli_lines = [] 

102 continue 

103 

104 if command == "@@python": 

105 self.write_cli_request( 

106 run_id, 

107 { 

108 "type": "@@python", 

109 "lines": cli_lines 

110 } 

111 ) 

112 is_waiting_for_response = True 

113 cli_lines = [] 

114 continue 

115 

116 if command.startswith("@@mode"): 

117 cmds = command.split(" ") 

118 if len(cmds) != 2 or cmds[1] not in ("off", "bash", "python"): 

119 print("Usage:") 

120 print("@@mode off") 

121 print("@@mode python") 

122 print("@@mode bash") 

123 else: 

124 line_mode = cmds[1] 

125 continue 

126 

127 if command.startswith("@@log"): 

128 cmds = command.split(" ") 

129 if len(cmds) != 2: 

130 print("Usage:") 

131 print("@@log <filename>") 

132 else: 

133 log_filename = cmds[1] 

134 continue 

135 

136 if command.startswith("@@load"): 

137 cmds = command.split(" ") 

138 if len(cmds) != 2: 

139 print("Usage:") 

140 print("@@load <filename>") 

141 else: 

142 try: 

143 with open(cmds[1], "rt") as load_f: 

144 for line in load_f: 

145 cli_lines.append(line.rstrip()) 

146 except Exception as e: 

147 print(f"Unable to read from file: {str(e)}") 

148 continue 

149 

150 if command == "@@clear": 

151 cli_lines = [] 

152 continue 

153 

154 if command == "@@show": 

155 for line in cli_lines: 

156 print(line) 

157 print() 

158 continue 

159 

160 # for any other command, we will append to the cli buffer 

161 if line_mode == "off": 

162 cli_lines.append(command) 

163 else: 

164 self.write_cli_request( 

165 run_id, 

166 { 

167 "type": "@@" + line_mode, 

168 "lines": [ command ] 

169 } 

170 ) 

171 is_waiting_for_response = True 

172 else: 

173 if self.has_cli_response(run_id): 

174 response = self.read_cli_response(run_id) 

175 # print('#################################################') 

176 # print('# Response #') 

177 # print(f"# status : {response['status']}") 

178 # if 'exit_code' in response: 

179 # print(f"# exit_code: {response['exit_code']}") 

180 # print('#################################################') 

181 cprint(response['output'], 'green', 'on_red') 

182 if log_filename is not None: 

183 with open(log_filename, "a+t") as log_f: 

184 print(response['output'], file=log_f) 

185 print("", file=log_f) 

186 is_waiting_for_response = False 

187 if command == "@@quit": 

188 break 

189 else: 

190 time.sleep(1) # do not sleep too long since this is an interactive session 

191 print(f"\r{cli_wait_prompt[cli_wait_prompt_idx]}\r", end="") 

192 cli_wait_prompt_idx = (cli_wait_prompt_idx + 1) % 4

299 r = os_client.list_objects(
